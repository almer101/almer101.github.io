def simulate_BM(dt, B):
    max_iter = 10000
    w = [0]
    while w[-1] < B:
        dw = sigma * np.sqrt(dt) * np.random.normal(0,1)
        w.append(w[-1] + dw)
        if len(w) > max_iter:
            return ([0],[0])

    ts = np.linspace(0, dt * (len(w)-1), len(w))
    return (ts, w)
